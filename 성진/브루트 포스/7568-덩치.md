## 7568 덩치

### 문제링크
https://www.acmicpc.net/problem/7568

### 문제풀이 및 주석

```C++
#include <iostream>
#include <string>
using namespace std;

struct dungchi
{
	int height;
	int weight;
};
int main()
{
	int n;
	cin >> n;
	dungchi* dungchiarr = new dungchi[n];
	int* dungchiscore = new int[n];
	for (int i = 0; i < n; i++)
	{
		dungchiscore[i] = 0;
		cin >> dungchiarr[i].height>>dungchiarr[i].weight;
	}

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (dungchiarr[i].height < dungchiarr[j].height && dungchiarr[i].weight < dungchiarr[j].weight)
				dungchiscore[i]++;
		}
	}


	for (int i = 0; i < n; i++)
		cout << dungchiscore[i]+1 << endl;
}
```
